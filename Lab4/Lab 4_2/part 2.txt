int printArray(int arr[], int length) // function to print array
{
    for (int i = 0; i < length; i++) 
    {     
        printf("%d ", arr[i]);     
    }    
}

int main() //main function 
{        
    int arr[] = {100, 200, 300, 400};   
    int length = sizeof(arr)/sizeof(arr[0]);    
    printArray(arr, length);         
    
    return 0;    
}
